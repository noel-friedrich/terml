DEF "PYRAMID" "height"
                NEW "count"
                NEW "halfheight" height
                DIV halfheight 2
                FLOOR halfheight
                REPEAT height
                    ADD count 1
                    NEW "padcount" halfheight
                    NEW "padsub" count
                    DIV padsub 2
                    FLOOR padsub
                    SUB padcount padsub

                    REPEAT padcount
                        OUT "  "
                    REPEAT count
                        OUT "**"
                    OUT "\n"

            PYRAMID 40
